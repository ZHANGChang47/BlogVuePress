# 全排列与子集

# 1、全排列

package com.chang;
import java.util.*;
public class test2 {
    public static void main(String[] args) {
        String[] str={"a","b","c"};
        final List<List<String>> subsets = subsets(str);
        System.out.println(subsets);
    }
    public static List<List<String>> subsets(String[] str){
        List<List<String>> ans=new ArrayList<>();
        int len=str.length;
        if(len==0){
            return ans;
        }
        Deque<String> deque=new ArrayDeque<>();
        boolean[] used=new boolean[len];	//使用一个used布尔数组用来标记已经可以使用的元素，若标记为false则可以使用
        dfs(str,0,len,deque,ans,used);
        return ans;
    }
    private static void dfs(String[] str, int index, int len, Deque<String> deque, List<List<String>> ans,boolean[] used) {
        if(index==len){
            ans.add(new ArrayList<>(deque));
            return;
        }
        for(int i=0;i<len;i++){
            if(used[i]){
                continue;	//表示已经用过
            }
            deque.addLast(str[i]);
            used[i]=true;
            dfs(str,index+1,len,deque,ans,used);	//进行下一论循环
            deque.removeLast();
            used[i]=false;
        }
    }
}
/*
[[a, b, c], [a, c, b], [b, a, c], [b, c, a], [c, a, b], [c, b, a]]
*/

# 3、含重复元素

如果参数数组中有重复元素，则应该增加新的判别条件，并放在if(used[i])之后：

// 剪枝条件：i > 0 是为了保证 nums[i - 1] 有意义
// 写 !used[i - 1] 是因为 nums[i - 1] 在深度优先遍历的过程中刚刚被撤销选择
if(i>0 && nums[i]==nums[i-1] && !used[i-1]){
	continue;
}

class Solution {
    public List<List<Integer>> permuteUnique(int[] nums) {
        int len=nums.length;
        List<List<Integer>> res=new ArrayList<>();
        if(len==0){
            return res;
        }
        // 排序（升序或者降序都可以），排序是剪枝的前提
        Arrays.sort(nums);
        boolean[] used=new boolean[len];
        // 使用 Deque 是 Java 官方 Stack 类的建议
        Deque<Integer> path=new ArrayDeque<>(len);
        dfs(nums,len,0,res,used,path);
        return res;
    }
    public void dfs(int[] nums,int len,int depth,List<List<Integer>> res,boolean[] used,Deque<Integer> path){
        if(len==depth){
            res.add(new ArrayList<>(path));
        }
        for(int i=0;i<len;i++){
            if(used[i]){
                continue;
            }
            // 剪枝条件：i > 0 是为了保证 nums[i - 1] 有意义
            // 写 !used[i - 1] 是因为 nums[i - 1] 在深度优先遍历的过程中刚刚被撤销选择
            if(i>0 && nums[i]==nums[i-1] && !used[i-1]){
                continue;
            }
            path.addLast(nums[i]);
            used[i]=true;
            dfs(nums,len,depth+1,res,used,path);
            // 回溯部分的代码，和 dfs 之前的代码是对称的
            path.removeLast();
            used[i]=false;
        }
    }
}

# 4、路径查找

79. 单词搜索 - 力扣（LeetCode） (leetcode-cn.com) (opens new window)

class Solution {
    private static final int[][] DIRECTIONS = {{-1, 0}, {0, -1}, {0, 1}, {1, 0}};
    private int rows;
    private int cols;
    private int len;
    private boolean[][] visited;
    private char[] charArray;
    private char[][] board;
    public boolean exist(char[][] board, String word) {
        rows = board.length;
        if (rows == 0) {
            return false;
        }
        cols = board[0].length;
        visited = new boolean[rows][cols];
        this.len = word.length();
        this.charArray = word.toCharArray();
        this.board = board;
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                if (dfs(i, j, 0)) {
                    return true;
                }
            }
        }
        return false;
    }
    private boolean dfs(int x, int y, int begin) {
        if (begin == len - 1) {
            return board[x][y] == charArray[begin];
        }
        if (board[x][y] == charArray[begin]) {
            visited[x][y] = true;
            for (int[] direction : DIRECTIONS) {
                int newX = x + direction[0];
                int newY = y + direction[1];
                if (inArea(newX, newY) && !visited[newX][newY]) {
                    if (dfs(newX, newY, begin + 1)) {
                        return true;
                    }
                }
            }
            visited[x][y] = false;
        }
        return false;
    }
    private boolean inArea(int x, int y) {
        return x >= 0 && x < rows && y >= 0 && y < cols;
    }
}
